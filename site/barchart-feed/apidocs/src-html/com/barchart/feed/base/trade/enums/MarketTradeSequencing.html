008    package com.barchart.feed.base.trade.enums;
009    
010    import com.barchart.util.values.api.Value;
011    
012    public enum MarketTradeSequencing implements Value<MarketTradeSequencing> {
013    
014            // Null value
015            NULL_TRADE_SEQUENCE,
016    
017            // Regular in-sequence trade
018            NORMAL,
019    
020            // Out-of-sequence trade, may update current high/low but not last
021            UNSEQUENCED,
022    
023            ;
024    
025            @Override
026            public MarketTradeSequencing freeze() {
027                    return this;
028            }
029    
030            @Override
031            public boolean isFrozen() {
032                    return true;
033            }
034    
035            @Override
036            public boolean isNull() {
037                    return this == NULL_TRADE_SEQUENCE;
038            }
039    
040    }
