008    package com.barchart.feed.base.trade.enums;
009    
010    import static com.barchart.feed.base.trade.enums.MarketTradeSequencing.NORMAL;
011    import static com.barchart.feed.base.trade.enums.MarketTradeSequencing.UNSEQUENCED;
012    import static com.barchart.feed.base.trade.enums.MarketTradeSession.COMBINED;
013    import static com.barchart.feed.base.trade.enums.MarketTradeSession.DEFAULT;
014    import static com.barchart.feed.base.trade.enums.MarketTradeSession.EXTENDED;
015    import static com.barchart.feed.base.trade.enums.MarketTradeSession.PIT;
016    
017    import com.barchart.util.values.api.Value;
018    
019    public enum MarketTradeType implements Value<MarketTradeType> {
020    
021            // Null trade type
022            NULL_TRADE_TYPE(DEFAULT, NORMAL),
023    
024            // Unknown trade type
025            UNKNOWN(DEFAULT, NORMAL),
026    
027            /** FUTURES */
028    
029            // Composite session (pit + electronic)
030            FUTURE_COMPOSITE(COMBINED, NORMAL),
031            // Electronic/overnight session
032            FUTURE_ELECTRONIC(DEFAULT, NORMAL),
033            // Pit / open outcry session
034            FUTURE_PIT(PIT, NORMAL),
035    
036            /** EQUITIES */
037    
038            // Regular trade types
039            REGULAR_SALE(DEFAULT, NORMAL),
040            ACQUISITION(DEFAULT, NORMAL),
041            AMEX_RULE_155(DEFAULT, NORMAL),
042            AUTOMATIC_EXECUTION(DEFAULT, NORMAL),
043            BUNCHED_TRADE(DEFAULT, NORMAL),
044            CROSS_TRADE(DEFAULT, NORMAL),
045            DISTRIBUTION(DEFAULT, NORMAL),
046            INTERMARKET_SWEEP(DEFAULT, NORMAL),
047            MARKET_CLOSING(DEFAULT, NORMAL),
048            MARKET_OPENING(DEFAULT, UNSEQUENCED),
049            MARKET_REOPENING(DEFAULT, NORMAL),
050            NYSE_RULE_127(DEFAULT, NORMAL),
051            RESERVED(DEFAULT, NORMAL),
052            SPLIT(DEFAULT, NORMAL),
053            STOCK_OPTION(DEFAULT, NORMAL),
054            STOPPED_STOCK_REGULAR(DEFAULT, NORMAL),
055            YELLOW_FLAG(DEFAULT, NORMAL),
056    
057            // Regular out-of-sequence trade types
058            BUNCHED_SOLD(DEFAULT, UNSEQUENCED),
059            DERIVATIVELY_PRICED(DEFAULT, UNSEQUENCED),
060            PRIOR_REFERENCE_PRICE(DEFAULT, UNSEQUENCED),
061            SOLD_LAST(DEFAULT, UNSEQUENCED),
062            SOLD_OOO(DEFAULT, UNSEQUENCED),
063            STOPPED_STOCK_OOO(DEFAULT, UNSEQUENCED),
064            STOPPED_STOCK_SOLD_LAST(DEFAULT, UNSEQUENCED),
065    
066            // Form-T (pre/post-market) trades
067            FORM_T(EXTENDED, NORMAL),
068            FORM_T_OOO(EXTENDED, UNSEQUENCED),
069    
070            ;
071    
072            public final byte ord;
073            public final MarketTradeSession session;
074            public final MarketTradeSequencing sequencing;
075    
076            private final static MarketTradeType[] ENUM_VALUES = values();
077            private final static int ENUM_SIZE = ENUM_VALUES.length;
078    
079            public static final int size() {
080                    return ENUM_SIZE;
081            }
082    
083            private MarketTradeType(final MarketTradeSession se,
084                            final MarketTradeSequencing sq) {
085                    ord = (byte) ordinal();
086                    session = se;
087                    sequencing = sq;
088            }
089    
090            @Override
091            public MarketTradeType freeze() {
092                    return this;
093            }
094    
095            @Override
096            public boolean isFrozen() {
097                    return true;
098            }
099    
100            @Override
101            public boolean isNull() {
102                    return this == NULL_TRADE_TYPE;
103            }
104    
105    }
