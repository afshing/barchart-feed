015    /**
016     * Provides a callback method for changes in the 'logged-in' state of a data
017     * feed. Users wishing to perform actions on login or log off should instantiate
018     * this interface and bind it to a feed client.
019     * <p>
020     * Note: A feed state event will only fire when the state changes. If a state
021     * listener is bound to a feed after login, no login state event will fire.
022     * 
023     */
024    public interface FeedStateListener {
025    
026            public void stateUpdate(FeedState state);
027    
028    }
